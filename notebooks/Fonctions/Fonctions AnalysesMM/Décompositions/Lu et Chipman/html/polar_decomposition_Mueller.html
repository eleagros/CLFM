if strcmp(decomposition_type, 'forward')
    muel=muel;

    muel=muel.';
end

================= Propriétés de physicité de la matrice de Mueller ================= %%

[physicity_check,~,~,~]=physical_crit(muel);
physical_criterion=physicity_check;

================= Extraction de la matrice de diatténuation ================= %%

dvec=[muel(1,2),muel(1,3),muel(1,4)]*(1/muel(1,1));
D=((muel(1,2)^2+muel(1,3)^2+muel(1,4)^2)^0.5);
D1=(1-D^2)^0.5;

if D==0
    muel_0=muel;
    MD=eye(4);
else
    MD=[1,dvec;
        dvec',D1*eye(3)+(1-D1)*(dvec')*dvec/D^2];
    muel_0=muel/MD;
end

================= Extraction de la matrice du déphaseur ================= %%

[U_R,S_R,V_R]=svd(muel_0(2:4,2:4));

% Modification de MR lorsque le déterminant de M est négatif %
% (Les signes de s1 et s2 doivent être ajustés en fonction du problème rencontré au niveau du déterminant de la matrice M)

s1 = 1;
s2 = 1;

if sign(det(muel))<0
    s = [s1 s2 -(s1 * s2)];
else
    s =[1 1 1];
end

mR=U_R*diag(s)*V_R';
MR=[1,zeros(1,3);
    zeros(3,1),mR];

================= Extraction de la matrice de dépolarisation ================= %%

Mdelta=real(muel_0*MR');

================= Choix de la décomposition ================= %%

if strcmp(decomposition_type, 'forward')
    MD=MD;
    MR=MR;
    Mdelta=Mdelta;
elseif strcmp(decomposition_type, 'reverse')
    MD=MD.';
    MR=MR.';
    Mdelta=Mdelta.';
end

================= Extraction des paramètres polarimétriques ================= %%

% Extraction des paramètres de diatténuation %

total_diattenuation=sqrt((MD(1,2)^2+MD(1,3)^2+MD(1,4)^2));
linear_diattenuation=sqrt(MD(1,2)^2+MD(1,3)^2);
circular_diattenuation=abs(MD(1,4));

% Extraction des paramètres du retard de phase (R : retard de phase total) %

argu= 0.5*(MR(2,2)+MR(3,3)+MR(4,4))-0.5;

if abs(argu)>1
    if argu>0
        R=acos(1);
    else
        R=acos(-1);
    end
else
    R=acosd(real(argu));
end

total_retardance=R;
indice_normalisation_retard=1/(2*sin(R));

% Composantes du vecteur retard

a1=indice_normalisation_retard*(MR(3,4)-MR(4,3));
a2=indice_normalisation_retard*(MR(4,2)-MR(2,4));
a3=indice_normalisation_retard*(MR(2,3)-MR(3,2));
retardance_vector=[1,a1,a2,a3]';

% Extraction des retards de phase linéaire et circulaire

linear_retardance=acosd(real(MR(4,4)));
circular_retardance=atand(real((MR(3,2)-MR(2,3))/(MR(3,3)+MR(2,2))));

Détermination de l'azimut du retard de phase linéaire entre 0° et 90° : orientation_linear_retardance

if total_retardance<0
    orientation_linear_retardance=0;
    orientation_linear_retardance_full=0;
else

    orientation_linear_retardance = 0.5*atand(real(MR(2,4)/MR(4,3)));
    if sign(orientation_linear_retardance)<0
        orientation_linear_retardance = 90-abs(orientation_linear_retardance);
    else
        orientation_linear_retardance = orientation_linear_retardance;
    end

    if sign(MR(2,4))>0
        orientation_linear_retardance_full=orientation_linear_retardance;
    else
        orientation_linear_retardance_full=90+orientation_linear_retardance;
    end

end
